  id INT AUTO_INCREMENT PRIMARY KEY,
  title VARCHAR(100) NOT NULL,
  genre VARCHAR(50),
  platform VARCHAR(50),
  release_date DATE,
  price DECIMAL(10,2),
  description TEXT,
  image_url VARCHAR(255)
);
